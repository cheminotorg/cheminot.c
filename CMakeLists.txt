CMAKE_MINIMUM_REQUIRED(VERSION 2.6)

PROJECT(Cheminot)
SET(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++11")
INCLUDE_DIRECTORIES(${CMAKE_CURRENT_BINARY_DIR}/lib/jsoncpp)
LINK_DIRECTORIES(${CMAKE_CURRENT_BINARY_DIR}/lib/jsoncpp/lib)
ADD_EXECUTABLE(cheminot "cheminot.cpp")
TARGET_LINK_LIBRARIES(cheminot sqlite3)
TARGET_LINK_LIBRARIES(cheminot jsoncpp)
