CMAKE_MINIMUM_REQUIRED(VERSION 2.6)

PROJECT(Cheminot)

SET (YAJL_MAJOR 2)
SET (YAJL_MINOR 1)
SET (YAJL_MICRO 1)

SET (YAJL_DIST_NAME "yajl-${YAJL_MAJOR}.${YAJL_MINOR}.${YAJL_MICRO}")

INCLUDE_DIRECTORIES(${CMAKE_CURRENT_BINARY_DIR}/lib/${YAJL_DIST_NAME}/include)
LINK_DIRECTORIES(${CMAKE_CURRENT_BINARY_DIR}/lib/${YAJL_DIST_NAME}/lib)

ADD_EXECUTABLE(cheminot "cheminot.c")
TARGET_LINK_LIBRARIES(cheminot yajl_s)
